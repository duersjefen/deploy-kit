# CCW Environment Variables Template
# ⚠️ DO NOT COMMIT ACTUAL TOKENS TO GIT ⚠️
# Copy these to your CCW environment configuration

# GitHub (for PR creation, releases)
# Get from: gh auth token
GITHUB_TOKEN=gho_YOUR_TOKEN_HERE

# Linear (for issue tracking)
# Get from: https://linear.app/settings/api
LINEAR_API_KEY=lin_api_YOUR_KEY_HERE

# npm (for publishing packages)
# Get from: cat ~/.npmrc | grep _authToken
NPM_TOKEN=npm_YOUR_TOKEN_HERE

# AWS (for deployments) - OPTIONAL
# Only needed if CCW will deploy to AWS
AWS_ACCESS_KEY_ID=AKIA_YOUR_KEY_HERE
AWS_SECRET_ACCESS_KEY=YOUR_SECRET_HERE
AWS_REGION=eu-west-1

# =============================================
# How to get your actual tokens:
# =============================================

# GitHub:
#   gh auth token

# npm:
#   cat ~/.npmrc | grep _authToken

# Linear:
#   Visit https://linear.app/settings/api
#   Create new API key

# =============================================
# Network Access Settings:
# =============================================
